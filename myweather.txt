create database myweather;
use myweather;


CREATE TABLE rain (
    city varchar(12) NOT NULL DEFAULT '',
    town varchar(12) NOT NULL DEFAULT '',
    townName varchar(12) NOT NULL DEFAULT '',
    attribute varchar(12) NOT NULL DEFAULT '',
    oneHour varchar(12) NOT NULL DEFAULT '',
    oneDay varchar(12) NOT NULL DEFAULT ''

) ENGINE=InnoDB DEFAULT CHARSET=utf8;
